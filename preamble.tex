% آیکون‌های ساده در عنوان جعبه‌ها
\def\awiconTip{$\star$}
\def\awiconWarn{$\triangle$}
\def\awiconInfo{$\circ$}
\def\awiconCheck{$\square$}
\def\awiconLink{$\rightarrow$}
\newcommand{\awicon}[1]{%
  \ifstrequal{#1}{tip}{\awiconTip}{%
  \ifstrequal{#1}{warn}{\awiconWarn}{%
  \ifstrequal{#1}{info}{\awiconInfo}{%
  \ifstrequal{#1}{check}{\awiconCheck}{%
  \ifstrequal{#1}{link}{\awiconLink}{\textbullet}%
  }}}}%
}
